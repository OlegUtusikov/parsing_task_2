S -> VAR BLOCK
BLOCK -> V
BLOCK -> V BLOCK
BLOCK -> EPS
V -> NAME COLON TYPE SEMICOLON
V -> NAME COMMA V
NAME -> LETTER NAME'
NAME' -> EPS
NAME' -> LETTER NAME'
NAME' -> DIGIT NAME'

/*
	VAR -> "var"
	TYPE -> "integer"
	TYPE -> "real"
	TYPE -> "boolean"
	TYPE -> "char"
	EPS -> ""
	COLON -> ":"
	SEMICOLON -> ";"
	COMMA -> ","
*/

FIRST(NAME') = {EPS, LETTER, DIGIT}
FIRST(NAME) = {LETTER}
FIRST(V) = {LETTER}
FIRST(BLOCK) = {EPS, LETTER}
FIRST(S) = {VAR}
