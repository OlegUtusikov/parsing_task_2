S -> VAR BLOCK
BLOCK -> V
BLOCK -> V BLOCK
BLOCK -> EPS
V -> NAME COLON TYPE SEMICOLON
V -> NAME COMMA V

/*
	VAR -> "var"
	TYPE -> "integer"
	TYPE -> "real"
	TYPE -> "boolean"
	TYPE -> "char"
	EPS -> ""
	COLON -> ":"
	SEMICOLON -> ";"
	COMMA -> ","
*/

FIRST(V) = {NAME}
FIRST(BLOCK) = {EPS, NAME}
FIRST(S) = {VAR}

FOLLOW(V) = {END, NAME}
FOLLOW(BLOCK) = {END}
FOLLOW(S) = {END}
