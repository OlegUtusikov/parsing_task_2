S -> VAR BLOCK END
BLOCK -> V SEMICOLON
BLOCK -> V SEMICOLON BLOCK
BLOCK -> EPS
V -> NAME COLON TYPE
V -> NAME COMMA V

/*
	VAR -> "var"
	TYPE -> "integer"
	TYPE -> "real"
	TYPE -> "boolean"
	TYPE -> "char"
	EPS -> ""
	COLON -> ":"
	SEMICOLON -> ";"
	COMMA -> ","
*/

FIRST(V) = {NAME}
FIRST(BLOCK) = {EPS, NAME}
FIRST(S) = {VAR}

FOLLOW(V) = {SEMICOLON}
FOLLOW(BLOCK) = {END}
FOLLOW(S) = {END}
